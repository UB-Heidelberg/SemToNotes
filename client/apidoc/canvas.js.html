xrx.canvas.render = function(canvas, affineTransform, callback) {
  var ctx = canvas.getContext('2d');
  ctx.save();
  ctx.fillStyle = 'white';
  ctx.fillRect(0, 0, canvas.width, canvas.height);
  ctx.beginPath();
  xrx.canvas.setTransform_(ctx, affineTransform);
  callback();
  ctx.closePath();
  ctx.restore();
