goog.provide('xrx.canvas.Image');



goog.require('xrx.canvas.Element');



xrx.canvas.Image = function(canvas, image) {

  goog.base(this, canvas);

  this.image_ = image;

  this.height_ = 0;

  this.width_ = 0;
};
goog.inherits(xrx.canvas.Image, xrx.canvas.Element);



xrx.canvas.Image.setHeight = function(image, height) {
  this.height_ = height;
};



xrx.canvas.Image.setWidth = function(image, width) {
  this.width_ = width;
};



xrx.canvas.Image.prototype.draw = function() {
  this.context_.drawImage(this.image_, 0, 0, this.width_, this.height_);
};



xrx.canvas.Image.create = function(canvas, image) {
  return new xrx.canvas.Image(canvas, image);
};
