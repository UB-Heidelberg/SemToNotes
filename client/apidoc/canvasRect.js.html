goog.provide('xrx.canvas.Rect');



goog.require('xrx.canvas.Element');



xrx.canvas.Rect = function(canvas) {

  goog.base(this, canvas);

  this.x_ = 0;

  this.y_ = 0;

  this.width_ = 0;

  this.height_ = 0;

  this.fill_ = {
    color: 'black'
  };
};
goog.inherits(xrx.canvas.Rect, xrx.canvas.Element);



xrx.canvas.Rect.setHeight = function(rect, height) {
  rect.height_ = height;
};



xrx.canvas.Rect.setWidth = function(rect, width) {
  rect.width_ = width;
};



xrx.canvas.Rect.setFill = function(rect, color) {
  rect.fill_.color = color;
};



xrx.canvas.Rect.prototype.draw = function() {
  this.context_.beginPath();
  this.context_.moveTo(this.x_, this.y_);
  this.context_.lineTo(this.x_, this.y_ + this.height_);
  this.context_.lineTo(this.x_ + this.width_, this.y_ + this.height_);
  this.context_.lineTo(this.x_ + this.width_, this.y_);
  this.context_.fillStyle = this.fill_.color;
  this.context_.fill();
  this.context_.closePath();
};



xrx.canvas.Rect.create = function(canvas) {
  return new xrx.canvas.Rect(canvas);
};
