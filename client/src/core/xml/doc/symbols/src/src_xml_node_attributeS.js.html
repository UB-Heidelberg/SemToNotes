 17 xrx.node.AttributeS = function(num, parent) {
 18   goog.base(this, xrx.node.ATTRIBUTE);
 19 
 20 
 21   this.num_ = num;
 22 
 23 
 24 
 25   this.parent_ = parent;
 26 };

 39 xrx.node.AttributeS.prototype.getToken = function() {
 40   var xml = this.getInstance().xml().substr(this.parent_.getOffset(),
 41       this.parent_.getLength());
 42   var loc = this.getStream().attribute(xml, this.num_);
 43   
 44   return new xrx.token.Attribute(this.getLabel(), loc.offset, loc.length);
 45 };

 49 xrx.node.AttributeS.prototype.getLabel = function() {
 50   var label = this.parent_.getLabel().clone();
 51   label.push(this.num_);
 52 
 53   return label;
 54 };

126 xrx.node.AttributeS.prototype.getName = function() {
127   var xml = this.getInstance().xml().substr(this.parent_.getOffset(),
128       this.parent_.getLength());
129   var stream = new xrx.stream(xml);
130   var loc = stream.attrName(xml, this.num_);
131 
132   return loc.xml(xml);
133 };

137 xrx.node.AttributeS.prototype.getNamespaceUri = function(prefix) {
138   var ns = this.getInstance().getIndex().getNamespace(
139       this.parent_.getToken(), prefix);
140 
141   return ns ? ns.uri : '';
142 };
